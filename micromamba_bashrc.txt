

# >>> mamba initialize >>>
# !! Contents within this block are managed by 'mamba init' !!
export MAMBA_EXE="/home/carla/bin/micromamba";
export MAMBA_ROOT_PREFIX="/home/carla/micromamba";
__mamba_setup="$("$MAMBA_EXE" shell hook --shell bash --prefix "$MAMBA_ROOT_PREFIX" 2> /dev/null)"
if [ $? -eq 0 ]; then
    eval "$__mamba_setup"
else
    if [ -f "/home/carla/micromamba/etc/profile.d/micromamba.sh" ]; then
        . "/home/carla/micromamba/etc/profile.d/micromamba.sh"
    else
        export  PATH="/home/carla/micromamba/bin:$PATH"  # extra space after export prevents interference from conda init
    fi
fi
unset __mamba_setup
# <<< mamba initialize <<<
